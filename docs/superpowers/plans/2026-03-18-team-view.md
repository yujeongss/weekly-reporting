### Task 2: Summary metrics — per-member progress bars

- [ ] **Step 1: Write helper `getMemberAvgProgress(memberId)`**

```javascript
function getMemberAvgProgress(memberId) {
  const tasks = getMemberData(memberId).tasks || [];
  if (!tasks.length) return null;
  const sum = tasks.reduce((a, t) => a + (t.overallProgress || 0), 0);
  return Math.round(sum / tasks.length);
}
```

- [ ] **Step 2: Build `renderTeamSummaryMetrics()` — progress bars section**

```javascript
function renderTeamSummaryMetrics() {
  const progressRows = state.meta.members.map(m => {
    const pct = getMemberAvgProgress(m.id);
    const display = pct === null ? '—' : pct + '%';

        ${bar}
        <span class="team-progress-pct">${display}</span>
      </div>`;

  return `<div class="team-progress-list">${progressRows}</div>`;
}

function getTeamCounts() {
  let totalTasks = 0, issueTasks = 0, totalDeploys = 0;
  for (const m of state.meta.members) {
    const d = getMemberData(m.id);
    totalTasks += (d.tasks || []).length;
    issueTasks += (d.tasks || []).filter(t => (t.issues || []).some(i => i.text)).length;
    totalDeploys += (d.deployments || []).length;
  }
  return { totalTasks, issueTasks, totalDeploys };
}
```

```javascript
function renderTeamCardView() {
  const cards = state.meta.members.map(m => {
    const data = getMemberData(m.id);
    const pct = getMemberAvgProgress(m.id);
    const issueCount = (data.tasks || []).filter(t => (t.issues||[]).some(i=>i.text)).length;
    const taskRows = (data.tasks || []).map(t =>
      `<div class="team-card-task">
        <span class="team-card-kpi">${esc(t.kpiProduct || '(미입력)')}</span>
        <span class="team-card-pct">${t.overallProgress || 0}%</span>
      </div>`
    ).join('') || '<div class="team-card-empty">과제 없음</div>';

    return `
      <div class="team-card" onclick="setActiveTab('${m.id}')">
        <div class="team-card-header">
          <span class="team-card-name">${esc(m.name)}</span>
          <span class="team-card-division">${esc(m.division || '')}</span>
          ${issueCount > 0 ? `<span class="team-card-issue-badge">⚠ ${issueCount}</span>` : ''}
        </div>
        <div class="team-card-tasks">${taskRows}</div>
        ${pct !== null ? `
          <div class="team-card-footer">
            <div class="team-progress-bg"><div class="team-progress-fill" style="width:${pct}%"></div></div>
            <span class="team-progress-pct">${pct}%</span>
          </div>` : ''}
      </div>`;
  }).join('');

  return `<div class="team-card-grid">${cards}</div>`;
}
```

```javascript
function renderTeamTableView() {
  const rows = [];
  for (const m of state.meta.members) {
    const data = getMemberData(m.id);
    const tasks = data.tasks || [];
    if (!tasks.length) {
      rows.push(`
        <tr class="team-table-row">
          <td class="team-td-member">${esc(m.name)}</td>
          <td colspan="5" style="color:var(--text-muted);font-size:13px;padding:10px 12px;">과제 없음</td>
        </tr>`);
      continue;
    }
    tasks.forEach((t, ti) => {
      const curText = (t.currentWeekItems||[]).filter(i=>i.text).map(i=>`• ${i.text}`).join('\n');
      const nxtText = (t.nextWeekItems||[]).filter(i=>i.text).map(i=>`• ${i.text}`).join('\n');
      const issueText = (t.issues||[]).filter(i=>i.text).map(i=>i.text).join('\n');
      rows.push(`
        <tr class="team-table-row ${issueText ? 'has-issue' : ''}">
          ${ti === 0 ? `<td class="team-td-member" rowspan="${tasks.length}">${esc(m.name)}</td>` : ''}
          <td class="team-td team-td-kpi"
              contenteditable="true"
              data-member="${m.id}" data-task="${t.id}" data-field="kpiProduct"
              onblur="updateTaskCEField(this)">${esc(t.kpiProduct||'')}</td>
          <td class="team-td team-td-activity">${esc(curText)}</td>
          <td class="team-td team-td-activity">${esc(nxtText)}</td>
          <td class="team-td team-td-pct">
            <input type="number" class="team-pct-input" min="0" max="100"
                   value="${t.overallProgress||0}"
                   onchange="updateTaskField('${m.id}','${t.id}','overallProgress',+this.value);renderMain();">
          </td>
          <td class="team-td team-td-issue ${issueText ? 'issue-cell' : ''}">${esc(issueText)}</td>
        </tr>`);
    });
  }
  return `
    <div class="team-table-wrap">
      <table class="team-table">
        <thead>
          <tr>
            <th>담당자</th><th>KPI / 제품</th><th>금주활동</th>
            <th>차주계획</th><th>진척율</th><th>이슈</th>
          </tr>
        </thead>
        <tbody>${rows.join('')}</tbody>
      </table>
    </div>`;
}
```
